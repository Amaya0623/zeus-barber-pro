// 🔥 DESTRUCTOR NUCLEAR DE CACHÉ PARA ACTUALIZACIONES INVISIBLES (CORREGIDO) function comprobarVersionNube() { const VERSION_ACTUAL = "1.50"; // Subimos la versión a 1.50 para forzar el arreglo en todos const versionGuardada = localStorage.getItem('zeus_version'); if (versionGuardada !== VERSION_ACTUAL) { console.log("¡Actualización detectada! Destruyendo caché..."); localStorage.setItem('zeus_version', VERSION_ACTUAL); if ('serviceWorker' in navigator) { navigator.serviceWorker.getRegistrations().then(function(registrations) { for(let registration of registrations) { registration.unregister(); } }); } if ('caches' in window) { caches.keys().then((names) => { names.forEach(name => caches.delete(name)); }); } setTimeout(() => { // 🚀 LA MAGIA: Rescatamos el link original (con el Kiosko) y solo le inyectamos la versión let urlActual = new URL(window.location.href); urlActual.searchParams.set('v', new Date().getTime()); window.location.href = urlActual.href; }, 1000); } }
